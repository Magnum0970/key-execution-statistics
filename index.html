<script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

<script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

<style>

  *, *::before, *::after { box-sizing: border-box; margin: 0; padding: 0; }

  body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif; background: #f5f5f3; color: #1a1a18; min-height: 100vh; }

  input, select, textarea, button { font-family: inherit; font-size: 13px; color: #1a1a18; background: #fff; border: 0.5px solid #ccc; border-radius: 6px; outline: none; transition: border-color 0.15s; }

  input:focus, select:focus, textarea:focus { border-color: #888; }

  button { cursor: pointer; padding: 6px 14px; background: #fff; border: 0.5px solid #bbb; border-radius: 6px; }

  button:hover { background: #f1f1ef; }

  select { padding: 4px 8px; }

  textarea { resize: vertical; }

  .card { background: #fff; border: 0.5px solid #ddd; border-radius: 12px; overflow: hidden; }

  .metric { background: #f1f1ef; border-radius: 8px; padding: 12px 14px; }

  .metric-label { font-size: 11px; color: #666; font-weight: 500; text-transform: uppercase; letter-spacing: 0.04em; margin-bottom: 4px; }

  .metric-value { font-size: 22px; font-weight: 500; }

  @media (max-width: 600px) { .grid-metrics { grid-template-columns: repeat(4, 1fr) !important; } }

<script type="text/babel">

const { useState, useMemo } = React;


const PRODUCTS = ["Marketo", "Adobe Journey Optimizer", "Both", "Other"];

const STATUSES = ["Ready for Assignment", "In Review", "Re-Assignment", "Scheduled", "In Progress", "On Hold"];

const STATUS_COLORS = {

  "Ready for Assignment": { bg: "#E6F1FB", text: "#0C447C", dot: "#378ADD" },

  "In Review":            { bg: "#EEEDFE", text: "#3C3489", dot: "#7F77DD" },

  "Re-Assignment":        { bg: "#FAEEDA", text: "#633806", dot: "#BA7517" },

  "Scheduled":            { bg: "#EAF3DE", text: "#27500A", dot: "#639922" },

  "In Progress":          { bg: "#E1F5EE", text: "#085041", dot: "#1D9E75" },

  "On Hold":              { bg: "#FCEBEB", text: "#791F1F", dot: "#E24B4A" },

};

const HEALTH_COLORS = {

  Green:  { bg: "#EAF3DE", text: "#27500A", dot: "#639922" },

  Yellow: { bg: "#FAEEDA", text: "#633806", dot: "#BA7517" },

  Red:    { bg: "#FCEBEB", text: "#791F1F", dot: "#E24B4A" },

};

const CSM_COLORS = ["#378ADD","#1D9E75","#7F77DD","#D85A30","#BA7517","#D4537E","#0F6E56","#533AB7","#E24B4A","#639922"];


const initCSMs = () => [

  { id: 1, name: "Parley", clients: [

    { id: 1, name: "Client 1", status: "In Progress", health: "Green", product: "Marketo", notes: "" },

    { id: 2, name: "Client 2", status: "In Progress", health: "Green", product: "Marketo", notes: "" },

    { id: 3, name: "Client 3", status: "In Progress", health: "Green", product: "Marketo", notes: "" },

    { id: 4, name: "Client 4", status: "In Progress", health: "Green", product: "Marketo", notes: "" },

  ]},

  { id: 2, name: "Karen", clients: [

    { id: 1, name: "Client 1", status: "In Progress", health: "Green", product: "Marketo", notes: "" },

    { id: 2, name: "Client 2", status: "In Progress", health: "Green", product: "Marketo", notes: "" },

    { id: 3, name: "Client 3", status: "In Progress", health: "Green", product: "Marketo", notes: "" },

    { id: 4, name: "Client 4", status: "In Progress", health: "Green", product: "Marketo", notes: "" },

  ]},

];


const nextId = arr => arr.length ? Math.max(...arr.map(x => x.id)) + 1 : 1;


function Avatar({ name, color, size = 36 }) {

  const initials = name.split(" ").map(w => w[0]).join("").slice(0, 2).toUpperCase() || "?";

  return (

    <div style={{ width: size, height: size, borderRadius: "50%", background: color + "22", display: "flex", alignItems: "center", justifyContent: "center", fontSize: size * 0.36, fontWeight: 500, color, flexShrink: 0, border: `1.5px solid ${color}44` }}>

      {initials}

    </div>

  );

}


function StatusBadge({ status }) {

  const c = STATUS_COLORS[status] || { bg: "#f1f1ef", text: "#444", dot: "#888" };

  return (

    <span style={{ background: c.bg, color: c.text, fontSize: 11, fontWeight: 500, padding: "3px 8px", borderRadius: 20, whiteSpace: "nowrap", display: "inline-flex", alignItems: "center", gap: 5 }}>

      <span style={{ width: 6, height: 6, borderRadius: "50%", background: c.dot, display: "inline-block" }} />

      {status}

    </span>

  );

}


function HealthDot({ health }) {

  const c = HEALTH_COLORS[health] || HEALTH_COLORS.Green;

  return <span style={{ width: 10, height: 10, borderRadius: "50%", background: c.dot, display: "inline-block", flexShrink: 0 }} title={health} />;

}


function MiniBar({ csms }) {

  const total = csms.reduce((s, c) => s + c.clients.length, 0) || 1;

  return (

    <div style={{ display: "flex", height: 6, borderRadius: 4, overflow: "hidden", gap: 1, marginTop: 8 }}>

      {csms.map((csm, i) => (

        <div key={csm.id} title={`${csm.name}: ${csm.clients.length}`} style={{ flex: csm.clients.length || 0.05, background: CSM_COLORS[i % CSM_COLORS.length] }} />

      ))}

    </div>

  );

}


function Dashboard() {

  const [csms, setCSMs] = useState(initCSMs);

  const [newCSMName, setNewCSMName] = useState("");

  const [filterStatus, setFilterStatus] = useState("All");

  const [filterHealth, setFilterHealth] = useState("All");

  const [activeTab, setActiveTab] = useState("all");

  const [expandedCSM, setExpandedCSM] = useState(null);


  const updateCSM = (csmId, fn) => setCSMs(prev => prev.map(c => c.id === csmId ? fn(c) : c));

  const updateClient = (csmId, clientId, field, val) => updateCSM(csmId, csm => ({ ...csm, clients: csm.clients.map(cl => cl.id === clientId ? { ...cl, [field]: val } : cl) }));

  const addClient = csmId => updateCSM(csmId, csm => ({ ...csm, clients: [...csm.clients, { id: nextId(csm.clients), name: "", status: "Ready for Assignment", health: "Green", product: "Marketo", notes: "" }] }));

  const removeClient = (csmId, clientId) => updateCSM(csmId, csm => ({ ...csm, clients: csm.clients.filter(cl => cl.id !== clientId) }));

  const addCSM = () => { if (!newCSMName.trim()) return; setCSMs(prev => [...prev, { id: nextId(prev), name: newCSMName.trim(), clients: [] }]); setNewCSMName(""); };

  const removeCSM = id => setCSMs(prev => prev.filter(c => c.id !== id));

  const renameCSM = (id, name) => updateCSM(id, c => ({ ...c, name }));


  const teamStats = useMemo(() => {

    const all = csms.flatMap(c => c.clients);

    const counts = {};

    STATUSES.forEach(s => counts[s] = 0);

    all.forEach(cl => { if (counts[cl.status] !== undefined) counts[cl.status]++; });

    return { total: all.length, ...counts, red: all.filter(cl => cl.health === "Red").length };

  }, [csms]);


  const filteredCSMs = useMemo(() => csms.map(csm => ({

    ...csm,

    clients: csm.clients.filter(cl =>

      (filterStatus === "All" || cl.status === filterStatus) &&

      (filterHealth === "All" || cl.health === filterHealth)

    )

  })).filter(csm => activeTab === "all" || csm.id === parseInt(activeTab)), [csms, filterStatus, filterHealth, activeTab]);


  const exportCSV = () => {

    const rows = [["CSM","Client","Status","Health","Product","Notes"]];

    csms.forEach(csm => csm.clients.forEach(cl => rows.push([csm.name, cl.name, cl.status, cl.health, cl.product, cl.notes.replace(/,/g,";")])));

    const csv = rows.map(r => r.map(v => `"${v}"`).join(",")).join("\n");

    const a = document.createElement("a"); a.href = "data:text/csv;charset=utf-8," + encodeURIComponent(csv); a.download = "csm_dashboard.csv"; a.click();

  };


  return (

    <div style={{ maxWidth: 1100, margin: "0 auto", padding: "0 0 2rem" }}>

      {/* Header */}

      <div style={{ background: "#fff", borderBottom: "0.5px solid #ddd", padding: "14px 20px", display: "flex", alignItems: "center", justifyContent: "space-between", flexWrap: "wrap", gap: 8, position: "sticky", top: 0, zIndex: 10 }}>

        <div>

          <div style={{ fontSize: 11, fontWeight: 500, color: "#D85A30", textTransform: "uppercase", letterSpacing: "0.08em", marginBottom: 2 }}>Customer Success</div>

          <h1 style={{ fontSize: 20, fontWeight: 500 }}>Key Execution Dashboard</h1>

        </div>

        <button onClick={exportCSV}>Export CSV</button>

      </div>


      <div style={{ padding: "1rem 1.25rem" }}>

        {/* Team summary */}

        <div style={{ marginBottom: "1rem" }}>

          <div style={{ fontSize: 11, fontWeight: 500, color: "#666", marginBottom: 10, textTransform: "uppercase", letterSpacing: "0.05em" }}>Team overview — {csms.length} CSMs</div>

          <div className="grid-metrics" style={{ display: "grid", gridTemplateColumns: "repeat(8, 1fr)", gap: 8, marginBottom: 12 }}>

            {[["Total", teamStats.total, "#D85A30"],["In Progress", teamStats["In Progress"], "#1D9E75"],["Ready", teamStats["Ready for Assignment"], "#378ADD"],["In Review", teamStats["In Review"], "#7F77DD"],["On Hold", teamStats["On Hold"], "#BA7517"],["Re-Assign", teamStats["Re-Assignment"], "#D4537E"],["Scheduled", teamStats["Scheduled"], "#639922"],["Red Accts", teamStats.red, "#E24B4A"]].map(([l,v,c]) => (

              <div key={l} className="metric"><div className="metric-label">{l}</div><div className="metric-value" style={{ color: c }}>{v}</div></div>

            ))}

          </div>

          <div style={{ fontSize: 12, color: "#666", marginBottom: 4 }}>Engagement load by CSM</div>

          <MiniBar csms={csms} />

          <div style={{ display: "flex", gap: 12, flexWrap: "wrap", marginTop: 6 }}>

            {csms.map((csm, i) => (

              <span key={csm.id} style={{ fontSize: 11, color: "#666", display: "flex", alignItems: "center", gap: 4 }}>

                <span style={{ width: 8, height: 8, borderRadius: "50%", background: CSM_COLORS[i % CSM_COLORS.length], display: "inline-block" }} />

                {csm.name} ({csm.clients.length})

              </span>

            ))}

          </div>

        </div>


        {/* Filters + tabs */}

        <div style={{ display: "flex", alignItems: "center", gap: 8, flexWrap: "wrap", marginBottom: "1rem", paddingBottom: "1rem", borderBottom: "0.5px solid #ddd" }}>

          <select value={filterStatus} onChange={e => setFilterStatus(e.target.value)}>

            <option value="All">All statuses</option>

            {STATUSES.map(s => <option key={s}>{s}</option>)}

          </select>

          <select value={filterHealth} onChange={e => setFilterHealth(e.target.value)}>

            <option value="All">All health</option>

            <option>Green</option><option>Yellow</option><option>Red</option>

          </select>

          <div style={{ flex: 1 }} />

          <div style={{ display: "flex", gap: 4, flexWrap: "wrap" }}>

            {[{ id: "all", label: "All CSMs" }, ...csms.map(c => ({ id: String(c.id), label: c.name }))].map(tab => (

              <button key={tab.id} onClick={() => setActiveTab(tab.id)} style={{ fontSize: 12, padding: "4px 10px", background: activeTab === tab.id ? "#f1f1ef" : "#fff", fontWeight: activeTab === tab.id ? 500 : 400 }}>

                {tab.label}

              </button>

            ))}

          </div>

        </div>


        {/* CSM cards */}

        <div style={{ display: "flex", flexDirection: "column", gap: "1rem" }}>

          {filteredCSMs.map((csm, ci) => {

            const srcCSM = csms.find(c => c.id === csm.id);

            const inProg = srcCSM.clients.filter(cl => cl.status === "In Progress").length;

            const redCount = srcCSM.clients.filter(cl => cl.health === "Red").length;

            const isExpanded = expandedCSM === csm.id || filteredCSMs.length === 1;

            return (

              <div key={csm.id} className="card">

                <div style={{ padding: "12px 16px", display: "flex", alignItems: "center", gap: 12, cursor: "pointer", borderBottom: isExpanded ? "0.5px solid #ddd" : "none", background: "#fff" }} onClick={() => setExpandedCSM(isExpanded && filteredCSMs.length > 1 ? null : csm.id)}>

                  <Avatar name={srcCSM.name || "?"} color={CSM_COLORS[ci % CSM_COLORS.length]} />

                  <div style={{ flex: 1, minWidth: 0 }}>

                    <input value={srcCSM.name} onChange={e => renameCSM(csm.id, e.target.value)} onClick={e => e.stopPropagation()} style={{ fontSize: 15, fontWeight: 500, border: "none", background: "transparent", padding: 0, width: "100%", outline: "none" }} placeholder="CSM name" />

                  </div>

                  <div style={{ display: "flex", gap: 6, alignItems: "center", flexShrink: 0, flexWrap: "wrap" }}>

                    <StatusBadge status="In Progress" />

                    <span style={{ fontSize: 12, fontWeight: 500, color: CSM_COLORS[ci % CSM_COLORS.length] }}>{inProg}</span>

                    {redCount > 0 && <span style={{ fontSize: 11, background: "#FCEBEB", color: "#791F1F", padding: "2px 7px", borderRadius: 20, fontWeight: 500 }}>{redCount} red</span>}

                    <span style={{ fontSize: 12, color: "#666" }}>{srcCSM.clients.length} accts</span>

                    <span style={{ fontSize: 12, color: "#aaa" }}>{isExpanded ? "▲" : "▼"}</span>

                  </div>

                </div>

                {isExpanded && (

                  <div>

                    {csm.clients.length === 0 && <div style={{ padding: 16, fontSize: 13, color: "#666" }}>No accounts match the current filter.</div>}

                    {csm.clients.map((cl, idx) => (

                      <div key={cl.id} style={{ padding: "12px 16px", borderBottom: idx < csm.clients.length - 1 ? "0.5px solid #eee" : "none" }}>

                        <div style={{ display: "flex", alignItems: "center", gap: 8, flexWrap: "wrap", marginBottom: 8 }}>

                          <HealthDot health={cl.health} />

                          <input value={cl.name} onChange={e => updateClient(csm.id, cl.id, "name", e.target.value)} style={{ fontSize: 13, fontWeight: 500, border: "none", background: "transparent", padding: 0, flex: "1 1 100px", minWidth: 80, outline: "none" }} placeholder="Client name" />

                          <select value={cl.status} onChange={e => updateClient(csm.id, cl.id, "status", e.target.value)}>

                            {STATUSES.map(s => <option key={s}>{s}</option>)}

                          </select>

                          <select value={cl.health} onChange={e => updateClient(csm.id, cl.id, "health", e.target.value)}>

                            <option>Green</option><option>Yellow</option><option>Red</option>

                          </select>

                          <select value={cl.product} onChange={e => updateClient(csm.id, cl.id, "product", e.target.value)}>

                            {PRODUCTS.map(p => <option key={p}>{p}</option>)}

                          </select>

                          <button onClick={() => removeClient(csm.id, cl.id)} style={{ fontSize: 12, padding: "3px 8px", color: "#888" }}>✕</button>

                        </div>

                        <textarea value={cl.notes} onChange={e => updateClient(csm.id, cl.id, "notes", e.target.value)} placeholder="Status update & next steps..." rows={2} style={{ width: "100%", fontSize: 12, padding: "6px 8px", background: "#f9f9f7", border: "0.5px solid #ddd", borderRadius: 6 }} />

                      </div>

                    ))}

                    <div style={{ padding: "10px 16px", display: "flex", justifyContent: "space-between", background: "#f9f9f7" }}>

                      <button onClick={() => addClient(csm.id)}>+ Add client</button>

                      <button onClick={() => removeCSM(csm.id)} style={{ color: "#888" }}>Remove CSM</button>

                    </div>

                  </div>

                )}

              </div>

            );

          })}

          <div style={{ display: "flex", gap: 8, alignItems: "center", marginTop: 4 }}>

            <input value={newCSMName} onChange={e => setNewCSMName(e.target.value)} onKeyDown={e => e.key === "Enter" && addCSM()} placeholder="New CSM name..." style={{ fontSize: 13, flex: 1, maxWidth: 240, padding: "6px 10px" }} />

            <button onClick={addCSM}>+ Add CSM</button>

          </div>

        </div>

      </div>

    </div>

  );

}


ReactDOM.createRoot(document.getElementById("root")).render(<Dashboard />);

</script>

</body>

</html>
